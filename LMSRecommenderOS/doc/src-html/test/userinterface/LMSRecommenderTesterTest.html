015    public class LMSRecommenderTesterTest {
016    
017            public static final int NUM_RECOMMENDER = 4;
018            
019            @Test
020            public void testRunEvaluater() {
021                    LMSRecommenderTester main = new LMSRecommenderTester("main/userinterface/examples.config");
022                    main.setVisible(true);
023                    try {
024                            Thread.sleep(1000);
025                            for (int i = NUM_RECOMMENDER - 1; i >= 0; i--) {
026                                    main.list.setSelectedIndex(i);
027                                    main.setExample(i);
028                                    Thread.sleep(1000);
029                            }
030                            Thread.sleep(1000);
031                            main.runEvaluater(EvalType.LEAVE_ONE_OUT);
032                            Thread.sleep(10000);
033                            while(!main.eval.isEnabled());
034                            main.runEvaluater(EvalType.HOLD_OUT);
035                            Thread.sleep(10000);
036                            while(!main.cVal.isEnabled());
037                    } catch (InterruptedException e) {
038                            e.printStackTrace();
039                    }
040            }
