022    public class RatingTest {
023    
024            Rating rating = new Rating();
025    
026            @Test
027            public void testMakeRating() {
028                    rating = new Rating();
029                    rating = new Rating(new ProfileDescription(IDGenerator.next(),
030                                    23, Degree.Informatica, "09003", 8, 2, 4), 3.0);
031            }
032    
033            @Test
034            public void testCalcRating() throws NoSuchFieldException, SecurityException {
035                    double eval = 3.0;
036                    int avgEval = 4;
037                    ProfileDescription profile = new ProfileDescription(IDGenerator.next(),
038                                    23, Degree.Informatica, "09003", 8, 2, avgEval);
039                    rating = new Rating(profile, eval);
040                    assertTrue(rating.getNormEval() == (2.5 / avgEval * eval));
041            }
042    
043            @Test
044            public void testEvaluateNull() throws NoSuchFieldException,
045                            SecurityException {
046                    double eval = 0.0;
047                    int avgEval = 4;
048                    ProfileDescription profile = new ProfileDescription(IDGenerator.next(),
049                                    23, Degree.Informatica, "09003", 8, 2, avgEval);
050                    rating = new Rating(profile, eval);
051                    assertTrue(rating.getNormEval() == (2.5 / avgEval * eval));
052            }
053            
054            @Test
055            public void testGetterSetter() {
056                    Method[] methods = LearningObject.class.getDeclaredMethods();
057                    StringBuilder sbMethods = new StringBuilder();
058                    for (Method method : methods) {
059                            method.setAccessible(true);
060                            sbMethods.append(method);
061                    }
062                    String strMethods = sbMethods.toString().toUpperCase();
063                    Field[] fields = LearningObject.class.getDeclaredFields();
064                    for (Field field : fields) {
065                            field.setAccessible(true);
066                            // Ignore private static fields && private static final fields
067                            if (field.getModifiers() != 10 && field.getModifiers() != 26) {
068                                    assertTrue(strMethods.contains(("get" + field.getName())
069                                                    .toUpperCase()));
070                                    assertTrue(strMethods.contains(("set" + field.getName())
071                                                    .toUpperCase()));
072                            }
073                    }
074            }
075    
076            @Test
077            public void testToString() {
078                    Field[] fields = rating.getClass().getDeclaredFields();
079                    String strFields = rating.toString();
080                    for (Field field : fields) {
081                            field.setAccessible(true);
082                            // Ignore private static fields && private static final fields
083                            if (field.getModifiers() != 10 && field.getModifiers() != 26) {
084                                    assertTrue(strFields.contains(field.getName()));
085                            }
086                    }
087            }
088    }
