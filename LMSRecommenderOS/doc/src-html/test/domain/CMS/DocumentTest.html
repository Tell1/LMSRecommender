028            @Test
029            public void testGetterSetter() {
030                    Method[] methods = LearningObject.class.getDeclaredMethods();
031                    StringBuilder sbMethods = new StringBuilder();
032                    for (Method method : methods) {
033                            method.setAccessible(true);
034                            sbMethods.append(method);
035                    }
036                    String strMethods = sbMethods.toString().toUpperCase();
037                    Field[] fields = LearningObject.class.getDeclaredFields();
038                    for (Field field : fields) {
039                            field.setAccessible(true);
040                            // Ignore private static fields && private static final fields
041                            if (field.getModifiers() != 10 && field.getModifiers() != 26) {
042                                    assertTrue(strMethods.contains(("get" + field.getName())
043                                                    .toUpperCase()));
044                                    assertTrue(strMethods.contains(("set" + field.getName())
045                                                    .toUpperCase()));
046                            }
047                    }
048            }
