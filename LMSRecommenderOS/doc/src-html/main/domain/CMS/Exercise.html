001    package main.domain.CMS;
002    
003    import jcolibri.cbrcore.Attribute;
004    
005    import main.domain.cases.DocumentDescription;
006    
007    /**
008     * Java Bean that represents a exercise in an e-learning environment.
009     * 
010     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
011     *         Mueller-Pettenpohl</a>
012     * @version 1.0
013     */
014    public class Exercise extends LearningObject {
015    
016            DocumentDescription profile;
017    
018            public Exercise() {
019            }
020    
021            public Exercise(Integer id, String name) {
022                    super();
023                    this.id = id;
024                    this.name = name;
025            }
026    
027            public DocumentDescription getProfile() {
028                    return profile;
029            }
030    
031            public void setProfile(DocumentDescription profile) {
032                    this.profile = profile;
033            }
034    
035            @Override
036            public Attribute getIdAttribute() {
037                    return new Attribute("id", this.getClass());
038            }
039    
040            @Override
041            public String toString() {
042                    return "Exercise [name=" + name + "]";
043            }
044    
045    }
