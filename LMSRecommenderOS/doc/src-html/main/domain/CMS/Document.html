001    package main.domain.CMS;
002    
003    import jcolibri.cbrcore.Attribute;
004    import main.domain.cases.DocumentDescription;
005    
006    /**
007     * Java Bean that represents a document in an e-learning environment.
008     * 
009     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
010     *         Mueller-Pettenpohl</a>
011     * @version 1.0
012     */
013    public class Document extends LearningObject {
014    
015            DocumentDescription profile;
016    
017            public Document() {
018            }
019    
020            public Document(Integer id, String name) {
021                    super();
022                    this.id = id;
023                    this.name = name;
024            }
025    
026            public DocumentDescription getProfile() {
027                    return profile;
028            }
029    
030            public void setProfile(DocumentDescription profile) {
031                    this.profile = profile;
032            }
033    
034            @Override
035            public Attribute getIdAttribute() {
036                    return new Attribute("id", this.getClass());
037            }
038    
039            @Override
040            public String toString() {
041                    return "Document [name=" + name + "]";
042            }
043    
044    }
