019    /**
020     * Specific implementation of CBRCaseBase to allow collaborative
021     * recommendations. These kind of recommendations are based in the past
022     * experieces of other users.<br>
023     * This case base manages cases composed by a description, a solution and a
024     * result. The description usually contains the information about the user that
025     * made the recommendation, the solution contains the information about the
026     * recommended item, and the result stores the rating value for the item (the
027     * value that the user assigns to an item after evaluating it).<br>
028     * This way, this case base implementation stores cases as a table:<br>
029     * <table border="1" cellspacing="0" cellpadding="2" width="90">
030     * <tr>
031     * <td></td>
032     * <td>Item1</td>
033     * <td>Item2</td>
034     * <td>Item3</td>
035     * <td>Item4</td>
036     * <td>Item5</td>
037     * <td>...</td>
038     * <td>ItemM</td>
039     * </tr>
040     * <tr>
041     * <td>User1</td>
042     * <td></td>
043     * <td>rating12</td>
044     * <td></td>
045     * <td>rating14</td>
046     * <td></td>
047     * <td></td>
048     * <td></td>
049     * </tr>
050     * <tr>
051     * <td>User2</td>
052     * <td>rating21</td>
053     * <td></td>
054     * <td>rating23</td>
055     * <td></td>
056     * <td></td>
057     * <td></td>
058     * <td>rating2N</td>
059     * </tr>
060     * <tr>
061     * <td>User3</td>
062     * <td></td>
063     * <td></td>
064     * <td>rating33</td>
065     * <td>rating34</td>
066     * <td></td>
067     * <td></td>
068     * <td>rating3N</td>
069     * </tr>
070     * <tr>
071     * <td>...</td>
072     * <td></td>
073     * <td></td>
074     * <td></td>
075     * <td></td>
076     * <td></td>
077     * <td></td>
078     * <td></td>
079     * </tr>
080     * <tr>
081     * <td>UserN</td>
082     * <td></td>
083     * <td>ratingN2</td>
084     * <td></td>
085     * <td></td>
086     * <td>ratingN5</td>
087     * <td></td>
088     * <td>ratingNN</td>
089     * </tr>
090     * </table>
091     * The values of the users column and the items row are the ids of the
092     * description and solution parts of the case. These ids must be integer values.
093     * The ratings are obtained from an attribute of the result part. <br>
094     * Note that these cases base allows having different cases with the same
095     * description (because each user can make several recommendations).
096     * <p>
097     * Collaborative recommenders can be separated into three steps:
098     * <ol>
099     * <li>Weight all users with respect to similarity with the active user.
100     * <li>Select a subset of users as a set of predictors.
101     * <li>Normalize ratings and compute a prediction from a weighted combination of
102     * selected neighbors' ratings.
103     * </ol>
104     * There are several ways to compute first step. And it must be implemented by
105     * subclassing this class as in PearsonMatrixCaseBase. The subclasses must
106     * implement the abstract methods defined here that return the similarity among
107     * neighbors. The similarity among users is stored through SimilarityTuple
108     * objects.
109     * <p>
110     * To allow an efficient comparison, ratings must be sorted by neighbors. In the
111     * above table this means that cases are organized by rows. And that is done
112     * internally in this class through the organizeByDescriptionId() method. Then,
113     * the ratings of each user can be accessed through the getRatingTuples()
114     * method. A RatingTuple object stores the id of the item and its rating (id of
115     * the solution and id of the result).
116     * <p>
117     * There are also 2 internal classes (CommonRatingTuple and
118     * CommonRatingsIterator) that allow to obtain efficiently the common ratings of
119     * two users. This will be used by subclasses when computing the neighbors
120     * similarity. The code of these classes is an adaptation of the one developed
121     * by Jerome Kelleher and Derek Bridge for the Collaborative Movie Recommender
122     * project at University College Cork (Ireland).
123     * <p>
124     * Note: Some changes were necessary to make this case base evaluable with the
125     * LMSEvaluator.
126     * 
127     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
128     *         Mueller-Pettenpohl</a>
129     * 
130     * @version 1.1
131     * 
132     */

253            @SuppressWarnings("unchecked")
254            /**
255             * Organizes cases by the description id.
256             */
257            private void organizeByDescriptionId() {
258                    org.apache.commons.logging.LogFactory.getLog(this.getClass()).info(
259                                    "Organizing cases");
260                    try {
261                            byDescriptionId = new HashMap<Integer, ArrayList<MatrixCaseBase.RatingTuple>>();
262                            descriptions = new HashMap<Integer, CaseComponent>();
263                            solutions = new HashMap<Integer, CaseComponent>();
264                            for (CBRCase c : cases) {
265                                    Integer descId = (Integer) c.getID();
266                                    ArrayList<MatrixCaseBase.RatingTuple> list = byDescriptionId
267                                                    .get(descId);
268                                    if (list == null) {
269                                            list = new ArrayList<MatrixCaseBase.RatingTuple>();
270                                            byDescriptionId.put(descId, list);
271                                            descriptions.put(descId, c.getDescription());
272                                    }
273                                    Integer solId = (Integer) c.getSolution().getIdAttribute()
274                                                    .getValue(c.getSolution());
275                                    solutions.put(solId, c.getSolution());
276                                    Number resId = (Number) value.getValue(c.getResult());
277                                    list.add(new MatrixCaseBase.RatingTuple(solId, resId
278                                                    .doubleValue()));
279                                    ProgressController.step(MatrixCaseBase.class);
280    
281                            }
282                            for (ArrayList<MatrixCaseBase.RatingTuple> list : byDescriptionId
283                                            .values())
284                                    java.util.Collections.sort(list);
285                    } catch (AttributeAccessException e) {
286                            org.apache.commons.logging.LogFactory.getLog(this.getClass())
287                                            .error(e, e);
288    
289                    }
290            }

329            /**
330             * Stores a rating for an item.
331             * 
332             * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
333             *         Mueller-Pettenpohl</a>
334             * 
335             * @version 1.1
336             */
337            public class RatingTuple implements Comparable<Object> {
338                    int solutionId;
339                    double rating;
340    
341                    public RatingTuple(int solutionId, double rating) {
342                            super();
343                            this.solutionId = solutionId;
344                            this.rating = rating;
345                    }
346    
347                    @Override
348                    public boolean equals(Object o) {
349                            return solutionId == ((RatingTuple) o).getSolutionId();
350                    }
351    
352                    @Override
353                    public int compareTo(Object o) {
354                            return solutionId - ((RatingTuple) o).solutionId;
355                    }
356    
357                    /**
358                     * @return Returns the rating.
359                     */
360                    public double getRating() {
361                            return rating;
362                    }
363    
364                    /**
365                     * @return Returns the solutionId.
366                     */
367                    public int getSolutionId() {
368                            return solutionId;
369                    }
370    
371                    @Override
372                    public String toString() {
373                            return "(Solution: " + solutionId + "->" + rating + ")";
374                    }
375            }

377            /**
378             * Stores the similarity among neighbors.
379             * 
380             * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
381             *         Mueller-Pettenpohl</a>
382             * 
383             * @version 1.1
384             * 
385             */
386            public class SimilarTuple implements Comparable<Object> {
387                    int similarId;
388                    int ratingId;
389                    double similarity;
390    
391                    public SimilarTuple(int similarId, double similarity) {
392                            super();
393                            this.similarId = similarId;
394                            this.similarity = similarity;
395                    }
396    
397                    /**
398                     * @return Returns the similarId.
399                     */
400                    public int getSimilarId() {
401                            return similarId;
402                    }
403    
404                    public int getRatingId() {
405                            return ratingId;
406                    }
407    
408                    /**
409                     * @return Returns the similarity.
410                     */
411                    public double getSimilarity() {
412                            return similarity;
413                    }
414    
415                    @Override
416                    public int compareTo(Object o) {
417                            return (int) (1000 * (((SimilarTuple) o).similarity - similarity));
418                    }
419    
420                    @Override
421                    public int hashCode() {
422                            final int prime = 31;
423                            int result = 1;
424                            result = prime * result + getOuterType().hashCode();
425                            result = prime * result + similarId;
426                            return result;
427                    }
428    
429                    @Override
430                    public boolean equals(Object obj) {
431                            if (this == obj)
432                                    return true;
433                            if (obj == null)
434                                    return false;
435                            if (getClass() != obj.getClass())
436                                    return false;
437                            SimilarTuple other = (SimilarTuple) obj;
438                            if (!getOuterType().equals(other.getOuterType()))
439                                    return false;
440                            if (similarId != other.similarId)
441                                    return false;
442                            return true;
443                    }
444    
445                    @Override
446                    public String toString() {
447                            return "[" + similarId + "," + similarity + "]";
448                    }
449    
450                    private MatrixCaseBase getOuterType() {
451                            return LMSMatrixCaseBase.this;
452                    }
453            }

517            /**
518             * Class that given two users returns an iterator over the common ratings.
519             * 
520             * 
521             */
522            class CommonRatingsIterator implements Iterator<Object> {
523                    int idA;
524                    int idB;
525                    Collection<MatrixCaseBase.RatingTuple> rtsA;
526                    Collection<MatrixCaseBase.RatingTuple> rtsB;
527                    Iterator<MatrixCaseBase.RatingTuple> iterA;
528                    Iterator<MatrixCaseBase.RatingTuple> iterB;
529    
530                    MatrixCaseBase.RatingTuple rka;
531                    MatrixCaseBase.RatingTuple rkb;
532                    boolean moreRatings;
533                    CommonRatingTuple current;
534    
535                    public CommonRatingsIterator(int idA, int idB,
536                                    Collection<MatrixCaseBase.RatingTuple> rtsA,
537                                    Collection<MatrixCaseBase.RatingTuple> rtsB) {
538                            super();
539                            this.idA = idA;
540                            this.idB = idB;
541                            this.rtsA = rtsA;
542                            this.rtsB = rtsB;
543    
544                            // initialise the iterators over the Correlatable's ratings
545                            iterA = rtsA.iterator();
546                            iterB = rtsB.iterator();
547                            if (iterA.hasNext() && iterB.hasNext()) {
548                                    rka = iterA.next();
549                                    rkb = iterB.next();
550                                    moreRatings = true; // MUST be before the first call to
551                                                                            // getNextMatch!
552                                    current = getNextMatch();
553                            } else {
554                                    current = null;
555                                    moreRatings = false;
556                            }
557                    }
558    
559                    /**
560                     * Return true iff there are more common ratings in this
561                     * CommonRatingsIterator.
562                     * 
563                     * @return true iff more matches exist; false otherwise.
564                     */
565                    @Override
566                    public boolean hasNext() {
567                            return current != null;
568                    }
569    
570                    /**
571                     * Return the next match in this CommonRatingsIterator. This is in the
572                     * form of a CommonRatingTuple. This allows both ratings concerned be
573                     * returned to the client code, thus saving a scan through the entire
574                     * ratings collection.
575                     * 
576                     * @return a RatingTuple of the next two common ratings.
577                     * @throws NoSuchElementException
578                     *             if no more common ratings exist.
579                     */
580                    @Override
581                    public Object next() throws NoSuchElementException {
582                            if (!hasNext()) {
583                                    throw new NoSuchElementException();
584                            }
585                            CommonRatingTuple temp = current;
586                            current = getNextMatch();
587                            return temp;
588                    }
589    
590                    /**
591                     * Throws UnsupportedOperationException if called.
592                     */
593                    @Override
594                    public void remove() throws UnsupportedOperationException {
595                            throw new UnsupportedOperationException("remove not supported in "
596                                            + "CommonRatingsIterator");
597                    }
598    
599                    /**
600                     * Return the next common rating between the two users. Return null if
601                     * no matches are left.
602                     * 
603                     * @return a RatingTuple of the next match in the two user collections
604                     *         of ratings; null if no matches exist.
605                     */
606                    private CommonRatingTuple getNextMatch() {
607                            CommonRatingTuple match = null;
608                            while (moreRatings && match == null) {
609                                    while (rka.compareTo(rkb) < 0 && moreRatings) {
610                                            if (iterA.hasNext()) {
611                                                    rka = iterA.next();
612                                            } else {
613                                                    moreRatings = false;
614                                            }
615                                    }
616                                    if (rka.compareTo(rkb) == 0) {
617                                            /*
618                                             * Match found.
619                                             */
620                                            match = new CommonRatingTuple(idA, rka.getRating(), idB,
621                                                            rkb.getRating(), rka.getSolutionId());
622    
623                                            if (iterA.hasNext() && iterB.hasNext()) {
624                                                    rka = iterA.next();
625                                                    rkb = iterB.next();
626                                            } else {
627                                                    moreRatings = false;
628                                            }
629                                    }
630                                    while (rka.compareTo(rkb) > 0 && moreRatings) {
631                                            if (iterB.hasNext()) {
632                                                    rkb = iterB.next();
633                                            } else {
634                                                    moreRatings = false;
635                                            }
636                                    }
637                            }
638                            return match;
639                    }
640            }
