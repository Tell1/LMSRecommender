001    package main.domain.cases;
002    
003    import jcolibri.cbrcore.Attribute;
004    import jcolibri.cbrcore.CaseComponent;
005    
006    /**
007     * Java Bean that stores the result of the case.
008     * 
009     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
010     *         Mueller-Pettenpohl</a>
011     * @version 1.0
012     */
013    public class RatingResult implements CaseComponent {
014    
015            Integer id;
016            Integer rating;
017    
018            public RatingResult() {
019            }
020            
021    
022            public RatingResult(Integer id, Integer rating) {
023                    this.id = id;
024                    this.rating = rating;
025            }
026    
027            public Integer getId() {
028                    return id;
029            }
030    
031            public void setId(Integer id) {
032                    this.id = id;
033            }
034    
035            public Integer getRating() {
036                    return rating;
037            }
038    
039            public void setRating(Integer rating) {
040                    this.rating = rating;
041            }
042    
043            @Override
044            public String toString() {
045                    return "Rating [id=" + id + ", rating=" + rating + "]";
046            }
047    
048            @Override
049            public Attribute getIdAttribute() {
050                    return new Attribute("id", RatingResult.class);
051            }
052    
053    }
