001    package main.domain;
002    
003    import jcolibri.cbraplications.StandardCBRApplication;
004    import jcolibri.cbrcore.CBRCaseBase;
005    import jcolibri.cbrcore.Connector;
006    import jcolibri.method.retrieve.NNretrieval.NNConfig;
007    
008    /**
009     * This abstract class provides the base structure of a CBR application for
010     * Learning Management Systems.
011     * 
012     * Defines the method of an standard CBR application. It is composed by:
013     * <ul>
014     * <li>A configuration method to set up the application.
015     * <li>A preCycle that loads cases and prepares the application to run.
016     * <li>The cycle method that runs a CBR step using the given query.
017     * <li>A postCycle in charge of finishing the application.
018     * </ul>
019     * 
020     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
021     *         Mueller-Pettenpohl</a>
022     * 
023     * @version 1.0
024     */
025    public abstract class StandardLMSRecommender implements StandardCBRApplication {
026    
027            /** */
028            private static StandardLMSRecommender instance = null;
029            /** */
030            protected Connector connector;
031            /** */
032            protected CBRCaseBase caseBase;
033            /** */
034            protected NNConfig simConfig;
035    
036            public static StandardLMSRecommender getInstance() {
037                    return instance;
038            }
039    
040            protected StandardLMSRecommender() {
041            }
042    
043            public Connector getConnector() {
044                    return connector;
045            }
046    
047            public void setConnector(Connector connector) {
048                    this.connector = connector;
049            }
050    
051            public CBRCaseBase getCaseBase() {
052                    return caseBase;
053            }
054    
055            public void setCaseBase(CBRCaseBase caseBase) {
056                    this.caseBase = caseBase;
057            }
058    
059            public NNConfig getSimConfig() {
060                    return simConfig;
061            }
062    
063            public void setSimConfig(NNConfig simConfig) {
064                    this.simConfig = simConfig;
065            }
066    
067    }
