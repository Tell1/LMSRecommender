040    /**
041     * Shows cases in a table allowing to select one.<br>
042     * This method is not very suitable for composed cases (cases with nested
043     * CaseComponents) because it doesn't reflects that structure.<br>
044     * It allways shows an "Add to Basket" and a "Quit" buttons. Optionally it shows
045     * an "Edit Query" button for conversational recommenders (not useful in
046     * one-shot recommenders).<br>
047     * 
048     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
049     *         Mueller-Pettenpohl</a>
050     * 
051     * @version 1.1
052     */
053    public class LMSDisplayCasesTableMethod extends DisplayCasesTableMethod {
054            private static JDialog dialog;
055    
056            private static ButtonGroup group;
057            private static int returnCode = UserChoice.QUIT;
058            private static CBRCase selectedCase = null;
059            private static JTable table;
060    
061            /** Shown cases */
062            private static CBRCase[] _cases;
063    
064            /**
065             * Display options for the methods. In BASIC mode the dialog only shows the
066             * buy and quit buttons. The EDIT_QUERY option returns a
067             * UserChoice.EDIT_QUERY value without any case selected. The SELECT_CASE
068             * option returns a UserChoice.EDIT_QUERY value but with a case selected
069             * from the list.
070             * 
071             * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
072             *         Mueller-Pettenpohl</a>
073             * 
074             * @version 1.1
075             * 
076             */
077            public enum DisplayOption {
078                    BASIC, EDIT_QUERY, SELECT_CASE
079            };
080    
081            /**
082             * Shows the dialog without the "Edit Query" option
083             * 
084             * @param cases
085             *            to display
086             * @return UserChoice object
087             */
088            public static UserChoice displayCasesInTableBasic(Collection<CBRCase> cases) {
089                    return displayCasesInTable(cases, DisplayOption.BASIC, null);
090            }
091    
092            /**
093             * Shows the dialog without the "Edit Query" option
094             * 
095             * @param cases
096             *            to display
097             * @return UserChoice object
098             */
099            public static UserChoice displayCasesInTableEditQuery(
100                            Collection<CBRCase> cases) {
101                    return displayCasesInTable(cases, DisplayOption.EDIT_QUERY,
102                                    "Refine Query");
103            }
104    
105            /**
106             * Shows the dialog without the "Edit Query" option
107             * 
108             * @param cases
109             *            to display
110             * @return UserChoice object
111             */
112            public static UserChoice displayCasesInTableSelectCase(
113                            Collection<CBRCase> cases) {
114                    return displayCasesInTable(cases, DisplayOption.SELECT_CASE,
115                                    "Something like this");
116            }
117    
118            /**
119             * Shows the dialog and allows to choose if show the "Edit Query" option.
120             * 
121             * @param cases
122             *            to display.
123             * @param displayOption
124             *            to set the options of the display
125             * @param optionLabel
126             *            to set the options of the label
127             * 
128             * @return UserChoice object.
129             */
130            static UserChoice displayCasesInTable(Collection<CBRCase> cases,
131                            DisplayOption displayOption, String optionLabel) {
132                    _cases = new CBRCase[cases.size()];
133                    cases.toArray(_cases);
134    
135                    dialog = new JDialog();
136                    dialog.setTitle(cases.size() + " Retrieved cases");
137                    dialog.setModal(true);
138    
139                    if (cases.size() == 0)
140                            return new UserChoice(UserChoice.REFINE_QUERY, selectedCase);
141    
142                    Vector<Object> columnNames = extractColumnNames(cases.iterator().next());
143    
144                    Vector<Object> rows = new Vector<Object>();
145                    for (CBRCase c : cases)
146                            rows.add(getAttributes(c));
147    
148                    table = new JTable(rows, columnNames) {
149    
150                            private static final long serialVersionUID = 1L;
151    
152                            @Override
153                            public void tableChanged(TableModelEvent e) {
154                                    super.tableChanged(e);
155                                    repaint();
156                            }
157    
158                    };
159    
160                    table.getColumn("Select").setCellRenderer(
161                                    new RadioButtonTableRenderer());
162                    table.getColumn("Select").setCellEditor(
163                                    new RadioButtonEditor(new JCheckBox()));
164    
165                    group = new ButtonGroup();
166                    TableModel tm = table.getModel();
167                    for (int i = 0; i < tm.getRowCount(); i++)
168                            group.add((JRadioButton) tm.getValueAt(i, 0));
169    
170                    JScrollPane scrollPane = new JScrollPane(table);
171                    table.setFillsViewportHeight(true);
172    
173                    JPanel mainPanel = new JPanel();
174                    mainPanel.setLayout(new BorderLayout());
175                    mainPanel.add(scrollPane, BorderLayout.CENTER);
176    
177                    JPanel actionsPanel = new JPanel();
178                    actionsPanel.setLayout(new BoxLayout(actionsPanel, BoxLayout.X_AXIS));
179    
180                    JButton ok = new JButton("Select Document");
181                    ok.addActionListener(new ActionListener() {
182                            @Override
183                            public void actionPerformed(ActionEvent arg0) {
184    
185                                    if (table.getSelectedRow() == -1)
186                                            JOptionPane.showMessageDialog(dialog,
187                                                            "You should choose one item", "Error",
188                                                            JOptionPane.ERROR_MESSAGE);
189                                    else {
190                                            returnCode = UserChoice.BUY;
191                                            selectedCase = _cases[table.getSelectedRow()];
192                                            dialog.setVisible(false);
193                                    }
194                            }
195                    });
196                    JButton quit = new JButton("Quit");
197                    quit.addActionListener(new ActionListener() {
198                            @Override
199                            public void actionPerformed(ActionEvent arg0) {
200                                    returnCode = UserChoice.QUIT;
201                                    dialog.setVisible(false);
202                            }
203                    });
204                    JButton refine = new JButton(optionLabel);
205                    refine.addActionListener(new ActionListener() {
206                            @Override
207                            public void actionPerformed(ActionEvent arg0) {
208    
209                                    returnCode = UserChoice.REFINE_QUERY;
210                                    dialog.setVisible(false);
211                            }
212                    });
213                    JButton select = new JButton(optionLabel);
214                    select.addActionListener(new ActionListener() {
215                            @Override
216                            public void actionPerformed(ActionEvent arg0) {
217                                    if (table.getSelectedRow() == -1)
218                                            JOptionPane.showMessageDialog(dialog,
219                                                            "You should choose one item", "Error",
220                                                            JOptionPane.ERROR_MESSAGE);
221                                    else {
222                                            returnCode = UserChoice.REFINE_QUERY;
223                                            selectedCase = _cases[table.getSelectedRow()];
224                                            dialog.setVisible(false);
225                                    }
226                            }
227                    });
228                    actionsPanel.add(Box.createHorizontalGlue());
229                    actionsPanel.add(ok);
230                    actionsPanel.add(quit);
231                    if (displayOption == DisplayOption.EDIT_QUERY)
232                            actionsPanel.add(refine);
233                    if (displayOption == DisplayOption.SELECT_CASE)
234                            actionsPanel.add(select);
235                    actionsPanel.add(Box.createHorizontalGlue());
236    
237                    mainPanel.add(actionsPanel, BorderLayout.SOUTH);
238    
239                    dialog.getContentPane().add(mainPanel);
240                    dialog.setSize(800, 200);
241                    jcolibri.method.gui.utils.WindowUtils.centerWindow(dialog);
242                    dialog.setVisible(true);
243    
244                    return new UserChoice(returnCode, selectedCase);
245            }

247            /**
248             * Returns a list with the values of the attributes of a case.
249             * 
250             * @param c
251             *            the case
252             * @return Vector of Objects
253             */
254            private static Vector<Object> getAttributes(CBRCase c) {
255                    Vector<Object> res = new Vector<Object>();
256    
257                    JRadioButton rb = new JRadioButton(c.getID().toString());
258                    res.add(rb);
259    
260                    getAttributes(c.getDescription(), res);
261                    getAttributes(c.getSolution(), res);
262                    getAttributes(c.getJustificationOfSolution(), res);
263                    getAttributes(c.getResult(), res);
264    
265                    return res;
266            }
267    
268            /**
269             * Fills the list with the values of the attributes of the CaseComponent.
270             * 
271             * @param cc
272             *            CaseComponent
273             * @param res
274             *            List to fill
275             */
276            private static void getAttributes(CaseComponent cc, Vector<Object> res) {
277                    Collection<Attribute> atts = AttributeUtils.getAttributes(cc);
278                    if (atts == null)
279                            return;
280    
281                    Attribute id = cc.getIdAttribute();
282                    for (Attribute a : atts) {
283                            if (!a.equals(id))
284                                    res.add(AttributeUtils.findValue(a, cc));
285                    }
286            }
287    
288            /**
289             * Obtains the column names of the tables from a case. (It obtains the names
290             * of the attributes)
291             * 
292             * @param c
293             *            is any case.
294             * @return a list of objects
295             */
296            private static Vector<Object> extractColumnNames(CBRCase c) {
297                    Vector<Object> res = new Vector<Object>();
298                    res.add("Select");
299                    extractColumnNames(c.getDescription(), res);
300                    extractColumnNames(c.getSolution(), res);
301                    extractColumnNames(c.getJustificationOfSolution(), res);
302                    extractColumnNames(c.getResult(), res);
303                    return res;
304            }
305    
306            /**
307             * Extracts the column names (names of the attributes) from a CaseComponent.
308             * 
309             * @param cc
310             *            is the CaseComponent.
311             * @param res
312             *            List to fill.
313             */
314            private static void extractColumnNames(CaseComponent cc, Vector<Object> res) {
315                    Collection<Attribute> atts = AttributeUtils.getAttributes(cc);
316                    if (atts == null)
317                            return;
318                    Attribute id = cc.getIdAttribute();
319                    for (Attribute a : atts) {
320                            if (!a.equals(id))
321                                    res.add(a.getName());
322                    }
323            }
