199            /**
200             * Runs a test in a different thread.
201             */
202            public void runExample() {
203                    run.setEnabled(false);
204                    try {
205                            int i = list.getSelectedIndex();
206                            if (i == -1) {
207                                    run.setEnabled(true);
208                                    return;
209                            }
210                            displayPane.setText("");
211    
212                            ExampleInfo ei = this.info.get(i);
213    
214                            MethodRunner mr = null;
215                            Class<?> c = ei.getMainClass();
216    
217                            Method[] methods = c.getMethods();
218                            for (Method m : methods)
219                                    if (m.getName().equals("main"))
220                                            mr = new MethodRunner(m);
221                            if (mr == null)
222                                    return;
223                            Thread executionThread = new Thread(mr);
224                            executionThread.start();
225    
226                    } catch (Exception e) {
227                            org.apache.commons.logging.LogFactory.getLog(this.getClass())
228                                            .error(e, e);
229                    }
230            }

232            public void runEvaluater(EvalType evalType) {
233                    eval.setEnabled(false);
234                    cVal.setEnabled(false);
235                    try {
236                            int i = list.getSelectedIndex();
237                            if (i == -1) {
238                                    eval.setEnabled(true);
239                                    cVal.setEnabled(true);
240                                    return;
241                            }
242                            displayPane.setText("");
243    
244                            ExampleInfo ei = this.info.get(i);
245    
246                            Method meth = ei.getMainClass().getDeclaredMethod("getInstance",
247                                            (Class<?>[]) null);
248                            Object obj = meth.invoke(null, (Object[]) null);
249                            StandardLMSRecommender newRec = null;
250                            if (obj instanceof StandardLMSRecommender) {
251                                    newRec = (StandardLMSRecommender) obj;
252                            }
253                            MethodRunnerWithArgs mr = null;
254                            Class<?> c = LMSRecommenderEvaluater.class;
255                            Method[] methods = c.getMethods();
256                            for (Method m : methods) {
257                                    if (m.getName().equals("evalRecommender")) {
258                                            mr = new MethodRunnerWithArgs(m, newRec, evalType);
259                                    }
260                            }
261                            if (mr == null)
262                                    return;
263                            Thread executionThread = new Thread(mr);
264                            executionThread.start();
265    
266                    } catch (Exception e) {
267                            org.apache.commons.logging.LogFactory.getLog(this.getClass())
268                                            .error(e, e);
269                    }
270            }

272            /**
273             * Thread for running a test
274             * 
275             * @author Juan A. Recio-Garcia
276             * @version 1.0
277             */
278            class MethodRunner implements Runnable {
279                    Method method;
280    
281                    MethodRunner(Method m) {
282                            this.method = m;
283                    }
284    
285                    @Override
286                    public void run() {
287                            Object[] args = new Object[1];
288                            args[0] = null;
289                            try {
290                                    method.invoke(null, args);
291                            } catch (Exception e) {
292                                    org.apache.commons.logging.LogFactory.getLog(this.getClass())
293                                                    .error(e);
294                                    e.printStackTrace();
295                            }
296                            org.apache.commons.logging.LogFactory.getLog(
297                                            LMSRecommenderTester.class).info("Test execution finished");
298                            run.setEnabled(true);
299                    }
300    
301            }

303            class MethodRunnerWithArgs implements Runnable {
304                    Method method;
305                    Object arg1;
306                    Object arg2;
307    
308                    MethodRunnerWithArgs(Method m, Object arg1, Object arg2) {
309                            this.method = m;
310                            this.arg1 = arg1;
311                            this.arg2 = arg2;
312                    }
313    
314                    @Override
315                    public void run() {
316                            try {
317                                    method.invoke(null, arg1, arg2);
318                            } catch (Exception e) {
319                                    org.apache.commons.logging.LogFactory.getLog(this.getClass())
320                                                    .error(e);
321                                    e.printStackTrace();
322                            }
323                            org.apache.commons.logging.LogFactory.getLog(
324                                            LMSRecommenderTester.class).info("Test execution finished");
325                            eval.setEnabled(true);
326                            cVal.setEnabled(true);
327                    }

330            /**
331             * Shows an example in the window.
332             */
333            public void setExample(int i) {
334                    try {
335                            ExampleInfo ei = this.info.get(i);
336                            tabPane.removeAll();
337    
338                            JEditorPane ep = new JEditorPane();
339                            ep.setContentType("text/html");
340                            ep.setText("<font face=\"verdana, arial, helvetica\"><b>"
341                                            + ei.getDescription() + "</b></font>");
342                            ep.setEditable(false);
343                            tabPane.add("Description", new JScrollPane(ep));
344    
345                            ep = new JEditorPane(ei.getSource());
346                            ep.setContentType("text/html");
347                            ep.setEditable(false);
348                            tabPane.add("Source", new JScrollPane(ep));
349    
350                            for (URL url : ei.getDoc()) {
351                                    int b = url.getFile().lastIndexOf('/') + 1;
352                                    int e = url.getFile().lastIndexOf(".html");
353                                    String name = url.getFile().substring(b, e);
354                                    if (name.equals("package-summary")) {
355                                            String aux = url.getFile().substring(0, b - 1);
356                                            int x = aux.lastIndexOf('/') + 1;
357                                            name = aux.substring(x);
358                                    }
359                                    tabPane.add("[Doc]" + name, new HTMLBrowser(url));
360                            }
361                    } catch (IOException e) {
362                            org.apache.commons.logging.LogFactory.getLog(this.getClass())
363                                            .error(e, e);
364                    }
365            }

377            /**
378             * Parses the information in the config file
379             */
380            public List<ExampleInfo> parseExampleInfo(String file) {
381                    ArrayList<ExampleInfo> info = new ArrayList<ExampleInfo>();
382    
383                    try {
384                            BufferedReader br = null;
385                            br = new BufferedReader(new InputStreamReader(FileIO.findFile(file)
386                                            .openStream()));
387                            String line = "";
388                            ExampleInfo ei;
389                            while ((line = br.readLine()) != null) {
390                                    ei = new ExampleInfo();
391                                    ei.setName(line);
392                                    ei.setDescription(br.readLine());
393                                    ei.setMainClass(Class.forName(br.readLine()));
394                                    ei.setSource(FileIO.findFile(br.readLine()));
395                                    while (!(line = br.readLine()).equals("<example>"))
396                                            ei.addDoc(FileIO.findFile(line));
397                                    info.add(ei);
398                            }
399                            br.close();
400                    } catch (Exception e) {
401                            org.apache.commons.logging.LogFactory.getLog(this.getClass())
402                                            .error(e, e);
403                    }
404                    return info;
405            }

514             * A HTML browser panel
515             * 
516             * @author Juan A. Recio-Garcia
517             * @version 1.0
518             * 
519             */
520            class HTMLBrowser extends JPanel {
521                    private static final long serialVersionUID = 1L;
522    
523                    JEditorPane doc;
524                    URL originalURL;
525    
526                    public HTMLBrowser(URL file) {
527                            originalURL = file;
528                            JPanel top = new JPanel();
529                            top.setLayout(new BoxLayout(top, BoxLayout.X_AXIS));
530    
531                            JButton prev = new JButton("Back");
532                            prev.addActionListener(new ActionListener() {
533    
534                                    @Override
535                                    public void actionPerformed(ActionEvent e) {
536                                            try {
537                                                    doc.setPage(originalURL);
538                                            } catch (IOException e1) {
539                                                    org.apache.commons.logging.LogFactory.getLog(
540                                                                    this.getClass()).error(e1, e1);
541                                            }
542                                    }
543                            });
544    
545                            JLabel urlInfo = new JLabel(file.toString());
546    
547                            top.add(prev);
548                            top.add(Box.createHorizontalGlue());
549                            top.add(urlInfo);
550    
551                            doc = new JEditorPane();
552                            doc.setContentType("text/html");
553                            doc.setEditable(false);
554                            doc.addHyperlinkListener(new HyperlinkListener() {
555    
556                                    @Override
557                                    public void hyperlinkUpdate(HyperlinkEvent e) {
558                                            try {
559                                                    if (e.getEventType().equals(
560                                                                    HyperlinkEvent.EventType.ACTIVATED))
561                                                            doc.setPage(e.getURL());
562    
563                                            } catch (IOException e1) {
564                                                    org.apache.commons.logging.LogFactory.getLog(
565                                                                    this.getClass()).error(e1, e1);
566                                            }
567                                    }
568    
569                            });
570                            try {
571                                    doc.setPage(file);
572                            } catch (IOException e) {
573                                    org.apache.commons.logging.LogFactory.getLog(this.getClass())
574                                                    .error(e, e);
575                            }
576    
577                            this.setLayout(new BorderLayout());
578                            this.add(top, BorderLayout.NORTH);
579                            this.add(new JScrollPane(doc), BorderLayout.CENTER);
580                    }
581    
582            }

584            /**
585             * This class redirects System.out to the log text area.
586             * 
587             * @author Juan A. Recio-Garcia
588             * @version 1.0
589             */
590            class FilteredStream extends FilterOutputStream {
591                    public FilteredStream(OutputStream aStream) {
592                            super(aStream);
593                    }
594    
595                    @Override
596                    public void write(byte b[]) throws IOException {
597                            String aString = new String(b);
598                            displayPane.append(aString);
599                            displayPane.setCaretPosition(displayPane.getDocument().getLength());
600                    }
601    
602                    @Override
603                    public void write(byte b[], int off, int len) throws IOException {
604                            String aString = new String(b, off, len);
605                            displayPane.append(aString);
606                            displayPane.setCaretPosition(displayPane.getDocument().getLength());
607                    }
608            }
