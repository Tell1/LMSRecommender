263            /**
264             * Process all the commands in the file (or stdin) associated with "this"
265             * object. Run SQL in the file through the given database connection.
266             * 
267             * This is synchronized so that I can use object variables to keep track of
268             * current line number, command, connection, i/o streams, etc.
269             * 
270             * Sets encoding character set to that specified with System Property
271             * 'sqlfile.charset'. Defaults to "US-ASCII".
272             * 
273             * @param conn
274             *            The JDBC connection to use for SQL Commands.
275             */
276            public synchronized void execute(Connection conn, PrintStream stdIn,
277                            PrintStream errIn, Boolean coeOverride) throws IOException,
278                            SqlToolError, SQLException {
279    
280                    psStd = stdIn;
281                    psErr = errIn;
282                    curConn = conn;
283                    curLinenum = -1;
284    
285                    String inputLine;
286                    String trimmedCommand;
287                    String trimmedInput;
288                    String deTerminated;
289                    boolean inComment = false; // Globbling up a comment
290                    int postCommentIndex;
291                    boolean gracefulExit = false;
292    
293                    continueOnError = (coeOverride == null) ? interactive : coeOverride
294                                    .booleanValue();
295    
296                    if (userVars != null && userVars.size() > 0) {
297                            plMode = true;
298                    }
299    
300                    String specifiedCharSet = System.getProperty("sqlfile.charset");
301    
302                    charset = ((specifiedCharSet == null) ? DEFAULT_CHARSET
303                                    : specifiedCharSet);
304    
305                    try {
306                            // br = new BufferedReader(new InputStreamReader((file == null)
307                            // ? System.in
308                            // : new FileInputStream(file), charset));
309                            br = new BufferedReader(new InputStreamReader(
310                                            jcolibri.util.FileIO.openFile(file.toString())));
311                            curLinenum = 0;
312    
313                            if (interactive) {
314                                    stdprintln(BANNER);
315                            }
316    
317                            while (true) {
318                                    if (interactive) {
319                                            psStd.print((stringBuffer.length() == 0) ? (chunking ? chunkPrompt
320                                                            : primaryPrompt)
321                                                            : contPrompt);
322                                    }
323    
324                                    inputLine = br.readLine();
325    
326                                    if (inputLine == null) {
327                                            /*
328                                             * This is because interactive EOD on some OSes doesn't send
329                                             * a line-break, resulting in no linebreak at all after the
330                                             * SqlFile prompt or whatever happens to be on their screen.
331                                             */
332                                            if (interactive) {
333                                                    psStd.println();
334                                            }
335    
336                                            break;
337                                    }
338    
339                                    curLinenum++;
340    
341                                    if (chunking) {
342                                            if (inputLine.equals(".")) {
343                                                    chunking = false;
344    
345                                                    setBuf(stringBuffer.toString());
346                                                    stringBuffer.setLength(0);
347    
348                                                    if (interactive) {
349                                                            stdprintln("Raw SQL chunk moved into buffer.  "
350                                                                            + "Run \":;\" to execute the chunk.");
351                                                    }
352                                            } else {
353                                                    if (stringBuffer.length() > 0) {
354                                                            stringBuffer.append('\n');
355                                                    }
356    
357                                                    stringBuffer.append(inputLine);
358                                            }
359    
360                                            continue;
361                                    }
362    
363                                    if (inComment) {
364                                            postCommentIndex = inputLine.indexOf("*/") + 2;
365    
366                                            if (postCommentIndex > 1) {
367    
368                                                    // I see no reason to leave comments in history.
369                                                    inputLine = inputLine.substring(postCommentIndex);
370    
371                                                    // Empty the buffer. The non-comment remainder of
372                                                    // this line is either the beginning of a new SQL
373                                                    // or Special command, or an empty line.
374                                                    stringBuffer.setLength(0);
375    
376                                                    inComment = false;
377                                            } else {
378    
379                                                    // Just completely ignore the input line.
380                                                    continue;
381                                            }
382                                    }
383    
384                                    trimmedInput = inputLine.trim();
385    
386                                    try {
387    
388                                            // This is the try for SQLException. SQLExceptions are
389                                            // normally thrown below in Statement processing, but
390                                            // could be called up above if a Special processing
391                                            // executes a SQL command from history.
392                                            if (stringBuffer.length() == 0) {
393                                                    if (trimmedInput.startsWith("/*")) {
394                                                            postCommentIndex = trimmedInput.indexOf("*/", 2) + 2;
395    
396                                                            if (postCommentIndex > 1) {
397    
398                                                                    // I see no reason to leave comments in
399                                                                    // history.
400                                                                    inputLine = inputLine
401                                                                                    .substring(postCommentIndex
402                                                                                                    + inputLine.length()
403                                                                                                    - trimmedInput.length());
404                                                                    trimmedInput = inputLine.trim();
405                                                            } else {
406    
407                                                                    // Just so we get continuation lines:
408                                                                    stringBuffer.append("COMMENT");
409    
410                                                                    inComment = true;
411    
412                                                                    continue;
413                                                            }
414                                                    }
415    
416                                                    // This is just to filter out useless newlines at
417                                                    // beginning of commands.
418                                                    if (trimmedInput.length() == 0) {
419                                                            continue;
420                                                    }
421    
422                                                    if (trimmedInput.charAt(0) == '*'
423                                                                    && (trimmedInput.length() < 2 || trimmedInput
424                                                                                    .charAt(1) != '{')) {
425                                                            try {
426                                                                    processPL((trimmedInput.length() == 1) ? ""
427                                                                                    : trimmedInput.substring(1).trim());
428                                                            } catch (BadSpecial bs) {
429                                                                    errprintln("Error at '"
430                                                                                    + ((file == null) ? "stdin" : file
431                                                                                                    .toString()) + "' line "
432                                                                                    + curLinenum + ":\n\"" + inputLine
433                                                                                    + "\"\n" + bs.getMessage());
434    
435                                                                    if (!continueOnError) {
436                                                                            throw new SqlToolError(bs);
437                                                                    }
438                                                            }
439    
440                                                            continue;
441                                                    }
442    
443                                                    if (trimmedInput.charAt(0) == '\\') {
444                                                            try {
445                                                                    processSpecial(trimmedInput.substring(1));
446                                                            } catch (BadSpecial bs) {
447                                                                    errprintln("Error at '"
448                                                                                    + ((file == null) ? "stdin" : file
449                                                                                                    .toString()) + "' line "
450                                                                                    + curLinenum + ":\n\"" + inputLine
451                                                                                    + "\"\n" + bs.getMessage());
452    
453                                                                    if (!continueOnError) {
454                                                                            throw new SqlToolError(bs);
455                                                                    }
456                                                            }
457    
458                                                            continue;
459                                                    }
460    
461                                                    if (trimmedInput.charAt(0) == ':'
462                                                                    && (interactive || (trimmedInput.charAt(1) == ';'))) {
463                                                            try {
464                                                                    processBuffer(trimmedInput.substring(1));
465                                                            } catch (BadSpecial bs) {
466                                                                    errprintln("Error at '"
467                                                                                    + ((file == null) ? "stdin" : file
468                                                                                                    .toString()) + "' line "
469                                                                                    + curLinenum + ":\n\"" + inputLine
470                                                                                    + "\"\n" + bs.getMessage());
471    
472                                                                    if (!continueOnError) {
473                                                                            throw new SqlToolError(bs);
474                                                                    }
475                                                            }
476    
477                                                            continue;
478                                                    }
479    
480                                                    String ucased = trimmedInput.toUpperCase();
481    
482                                                    if (ucased.startsWith("DECLARE")
483                                                                    || ucased.startsWith("BEGIN")) {
484                                                            chunking = true;
485    
486                                                            stringBuffer.append(inputLine);
487    
488                                                            if (interactive) {
489                                                                    stdprintln("Enter RAW SQL.  No \\, :, * commands.  "
490                                                                                    + "End with a line containing only \".\":");
491                                                            }
492    
493                                                            continue;
494                                                    }
495                                            }
496    
497                                            if (trimmedInput.length() == 0) {
498    
499                                                    // Blank lines delimit commands ONLY IN INTERACTIVE
500                                                    // MODE!
501                                                    if (interactive && !inComment) {
502                                                            setBuf(stringBuffer.toString());
503                                                            stringBuffer.setLength(0);
504                                                            stdprintln("Current input moved into buffer.");
505                                                    }
506    
507                                                    continue;
508                                            }
509    
510                                            deTerminated = deTerminated(inputLine);
511    
512                                            // A null terminal line (i.e., /\s*;\s*$/) is never useful.
513                                            if (!trimmedInput.equals(";")) {
514                                                    if (stringBuffer.length() > 0) {
515                                                            stringBuffer.append('\n');
516                                                    }
517    
518                                                    stringBuffer.append((deTerminated == null) ? inputLine
519                                                                    : deTerminated);
520                                            }
521    
522                                            if (deTerminated == null) {
523                                                    continue;
524                                            }
525    
526                                            // If we reach here, then stringBuffer contains a complete
527                                            // SQL command.
528                                            curCommand = stringBuffer.toString();
529                                            trimmedCommand = curCommand.trim();
530    
531                                            if (trimmedCommand.length() == 0) {
532                                                    throw new SQLException("Empty SQL Statement");
533                                            }
534    
535                                            setBuf(curCommand);
536                                            processSQL();
537                                    } catch (SQLException se) {
538                                            errprintln("SQL Error at '"
539                                                            + ((file == null) ? "stdin" : file.toString())
540                                                            + "' line " + curLinenum + ":\n\"" + curCommand
541                                                            + "\"\n" + se.getMessage());
542    
543                                            if (!continueOnError) {
544                                                    throw se;
545                                            }
546                                    } catch (BreakException be) {
547                                            String msg = be.getMessage();
548    
549                                            if ((!recursed) && (msg != null && !msg.equals("file"))) {
550                                                    errprintln("Unsatisfied break statement"
551                                                                    + ((msg == null) ? "" : (" (type " + msg + ')'))
552                                                                    + '.');
553                                            } else {
554                                                    gracefulExit = true;
555                                            }
556    
557                                            if (recursed || !continueOnError) {
558                                                    throw be;
559                                            }
560                                    } catch (ContinueException ce) {
561                                            String msg = ce.getMessage();
562    
563                                            if (!recursed) {
564                                                    errprintln("Unsatisfied continue statement"
565                                                                    + ((msg == null) ? "" : (" (type " + msg + ')'))
566                                                                    + '.');
567                                            } else {
568                                                    gracefulExit = true;
569                                            }
570    
571                                            if (recursed || !continueOnError) {
572                                                    throw ce;
573                                            }
574                                    } catch (QuitNow qn) {
575                                            throw qn;
576                                    } catch (SqlToolError ste) {
577                                            if (!continueOnError) {
578                                                    throw ste;
579                                            }
580                                    }
581    
582                                    stringBuffer.setLength(0);
583                            }
584    
585                            if (inComment || stringBuffer.length() != 0) {
586                                    errprintln("Unterminated input:  [" + stringBuffer + ']');
587    
588                                    throw new SqlToolError("Unterminated input:  [" + stringBuffer
589                                                    + ']');
590                            }
591    
592                            gracefulExit = true;
593                    } catch (QuitNow qn) {
594                            gracefulExit = qn.getMessage() == null;
595    
596                            if ((!recursed) && !gracefulExit) {
597                                    errprintln("Aborting: " + qn.getMessage());
598                            }
599    
600                            if (recursed || !gracefulExit) {
601                                    throw qn;
602                            }
603    
604                            return;
605                    } finally {
606                            closeQueryOutputStream();
607    
608                            if (fetchingVar != null) {
609                                    errprintln("PL variable setting incomplete:  " + fetchingVar);
610    
611                                    gracefulExit = false;
612                            }
613    
614                            if (br != null) {
615                                    br.close();
616                            }
617    
618                            if ((!gracefulExit) && possiblyUncommitteds.get()) {
619                                    errprintln("Rolling back SQL transaction.");
620                                    curConn.rollback();
621                                    possiblyUncommitteds.set(false);
622                            }
623                    }
624            }

626            /**
627             * Returns a copy of given string without a terminating semicolon. If there
628             * is no terminating semicolon, null is returned.
629             * 
630             * @param inString
631             *            Base String, which will not be modified (because a "copy" will
632             *            be returned).
633             */
634            private static String deTerminated(String inString) {

762

764                    int special;
765                    char commandChar = 'i';

767    
768                    if (inString.length() > 0) {
769                            commandChar = inString.charAt(0);
770                            other = inString.substring(1);
771    
772                            if (other.trim().length() == 0) {
773                                    other = null;
774                            }
775                    }
776

778

802                                            // complete SQL command.

832                            boolean modeGlobal = false;
833                            boolean modeExecute = false;

848                                            throw new BadSwitch(1);

863                                                            && !toker.nextToken().equals(delim)) {
864                                                    throw new BadSwitch(3);

869                                            String opts = toker.nextToken("");

900                                                                            "Unknown Substitution option: "
901                                                                                            + opts.charAt(j));
902                                                    }
903                                            }
904                                    }
905    
906                                    if (modeIC) {
907                                            fromHist = fromHist.toUpperCase();
908                                            from = from.toUpperCase();
909                                    }
910    
911                                    // lineStart will be either 0 or char FOLLOWING a \n.
912                                    int lineStart = 0;
913    
914                                    // lineStop is the \n AFTER what we consider.
915                                    int lineStop = -1;
916    
917                                    if (modeLine > 0) {
918                                            for (int j = 1; j < modeLine; j++) {
919                                                    lineStart = fromHist.indexOf('\n', lineStart) + 1;
920    
921                                                    if (lineStart < 1) {
922                                                            throw new BadSpecial("There are not " + modeLine
923                                                                            + " lines in the buffer.");
924                                                    }
925                                            }
926    
927                                            lineStop = fromHist.indexOf('\n', lineStart);
928                                    }
929    
930                                    if (lineStop < 0) {
931                                            lineStop = fromHist.length();
932                                    }
933    
934                                    // System.err.println("["
935                                    // + fromHist.substring(lineStart, lineStop) + ']');
936                                    int i;
937    
938                                    if (modeGlobal) {
939                                            i = lineStop;

941                                            while ((i = fromHist.lastIndexOf(from, i - 1)) >= lineStart) {
942                                                    sb.replace(i, i + from.length(), to);
943                                            }
944                                    } else if ((i = fromHist.indexOf(from, lineStart)) > -1
945                                                    && i < lineStop) {
946                                            sb.replace(i, i + from.length(), to);
947                                    }
948    
949                                    // statementHistory[curHist] = sb.toString();
950                                    curCommand = sb.toString();
951    
952                                    setBuf(curCommand);
953                                    stdprintln((modeExecute ? "Executing" : "Current Buffer")
954                                                    + ":\n" + curCommand);
955    
956                                    if (modeExecute) {
957                                            stdprintln();
958                                    }
959                            } catch (BadSwitch badswitch) {
960                                    throw new BadSpecial(
961                                                    "Substitution syntax:  \":s/from this/to that/i;g2\".  "
962                                                                    + "Use '$' for line separations.  ["
963                                                                    + badswitch.getMessage() + ']');
964                            }
965    
966                            if (modeExecute) {
967                                    processSQL();
968                                    stringBuffer.setLength(0);
969                            }
970    
971                            return;
972    
973                    case '?':
974                            stdprintln(BUFFER_HELP_TEXT);
975    
976                            return;
977                    }
978    
979                    throw new BadSpecial("Unknown Buffer Command");
980            }

1423                    }

1425                    throw new BadSpecial("Unknown Special Command");

1430            /**
1431             * Returns index specifying 1 past end of a variable name.
1432             * 
1433             * @param inString
1434             *            String containing a variable name
1435             * @param startIndex
1436             *            Index within inString where the variable name begins
1437             * @return Index within inString, 1 past end of the variable name
1438             */
1439            static int pastName(String inString, int startIndex) {
1440    
1441                    String workString = inString.substring(startIndex);
1442                    int e = inString.length(); // Index 1 past end of var name.
1443                    int nonVarIndex;
1444    
1445                    for (int i = 0; i < nonVarChars.length; i++) {
1446                            nonVarIndex = workString.indexOf(nonVarChars[i]);
1447    
1448                            if (nonVarIndex > -1 && nonVarIndex < e) {
1449                                    e = nonVarIndex;
1450                            }
1451                    }
1452    
1453                    return startIndex + e;
1454            }

1456            /**
1457             * Deference PL variables.
1458             * 
1459             * @throws SQLException
1460             *             This is really an inappropriate exception type. Only using it
1461             *             because I don't have time to do things properly.
1462             */
1463            @SuppressWarnings("unused")
1464            private String dereference(String inString, boolean permitAlias)
1465                            throws SQLException {
1466    
1467                    String varName, varValue;
1468                    StringBuffer expandBuffer = new StringBuffer(inString);
1469                    int b, e; // begin and end of name. end really 1 PAST name
1470                    int nonVarIndex;
1471    
1472                    if (permitAlias && inString.trim().charAt(0) == '/') {
1473                            int slashIndex = inString.indexOf('/');
1474    
1475                            e = pastName(inString.substring(slashIndex + 1), 0);
1476    
1477                            // In this case, e is the exact length of the var name.
1478                            if (e < 1) {
1479                                    throw new SQLException("Malformed PL alias use");
1480                            }
1481    
1482                            varName = inString.substring(slashIndex + 1, slashIndex + 1 + e);
1483                            varValue = (String) userVars.get(varName);
1484    
1485                            if (varValue == null) {
1486                                    throw new SQLException("Undefined PL variable:  " + varName);
1487                            }
1488    
1489                            expandBuffer.replace(slashIndex, slashIndex + 1 + e,
1490                                            (String) userVars.get(varName));
1491                    }
1492    
1493                    String s;
1494    
1495                    while (true) {
1496                            s = expandBuffer.toString();
1497                            b = s.indexOf("*{");
1498    
1499                            if (b < 0) {
1500    
1501                                    // No more unexpanded variable uses
1502                                    break;
1503                            }
1504    
1505                            e = s.indexOf('}', b + 2);
1506    
1507                            if (e == b + 2) {
1508                                    throw new SQLException("Empty PL variable name");
1509                            }
1510    
1511                            if (e < 0) {
1512                                    throw new SQLException("Unterminated PL variable name");
1513                            }
1514    
1515                            varName = s.substring(b + 2, e);
1516    
1517                            if (!userVars.containsKey(varName)) {
1518                                    throw new SQLException("Use of undefined PL variable: "
1519                                                    + varName);
1520                            }
1521    
1522                            expandBuffer.replace(b, e + 1, (String) userVars.get(varName));
1523                    }
1524    
1525                    return expandBuffer.toString();
1526            }

1710

1712                                    SqlFile sf;

1715                                            try {
1716                                                    varVal = values[i];

1766                                    throw new BadSpecial("Malformatted PL if command (1)");

1770    
1771                            if (parenExpr.length() < 2 || parenExpr.charAt(0) != '('
1772                                            || parenExpr.charAt(parenExpr.length() - 1) != ')') {

1949            /*
1950             * Read a PL block into a new temp file.
1951             * 
1952             * WARNING!!! foreach blocks are not yet smart about comments and strings.
1953             * We just look for a line beginning with a PL "end" command without
1954             * worrying about comments or quotes (for now).
1955             * 
1956             * WARNING!!! This is very rudimentary. Users give up all editing and
1957             * feedback capabilities while in the foreach loop. A better solution would
1958             * be to pass current input stream to a new SqlFile.execute() with a mode
1959             * whereby commands are written to a separate history but not executed.
1960             */
1961            private File plBlockFile(String type) throws IOException, SqlToolError {
1962    
1963                    String s;
1964                    StringTokenizer toker;
1965    
1966                    // Have already read the if/while/foreach statement, so we are already
1967                    // at nest level 1. When we reach nestlevel 1 (read 1 net "end"
1968                    // statement), we're at level 0 and return.
1969                    int nestlevel = 1;
1970                    String curPlCommand;
1971    
1972                    if (type == null
1973                                    || ((!type.equals("foreach")) && (!type.equals("if")) && (!type
1974                                                    .equals("while")))) {
1975                            throw new RuntimeException(
1976                                            "Assertion failed.  Unsupported PL block type:  " + type);
1977                    }
1978    
1979                    File tmpFile = File.createTempFile("sqltool-", ".sql");
1980                    PrintWriter pw = new PrintWriter(new OutputStreamWriter(
1981                                    new FileOutputStream(tmpFile), charset));
1982    
1983                    pw.println("/* " + (new java.util.Date()) + ". " + getClass().getName()
1984                                    + " PL block. */\n");
1985    
1986                    while (true) {
1987                            s = br.readLine();
1988    
1989                            if (s == null) {
1990                                    errprintln("Unterminated '" + type + "' PL block");
1991    
1992                                    throw new SqlToolError("Unterminated '" + type + "' PL block");
1993                            }
1994    
1995                            curLinenum++;
1996    
1997                            if (s.trim().length() > 1 && s.trim().charAt(0) == '*') {
1998                                    toker = new StringTokenizer(s.trim().substring(1));
1999                                    curPlCommand = toker.nextToken();
2000    
2001                                    // PL COMMAND of some sort.
2002                                    if (curPlCommand.equals(type)) {
2003                                            nestlevel++;
2004                                    } else if (curPlCommand.equals("end")) {
2005                                            if (toker.countTokens() < 1) {
2006                                                    errprintln("PL end statement requires arg of "
2007                                                                    + "'foreach' or 'if' or 'while' (1)");
2008    
2009                                                    throw new SqlToolError("PL end statement requires arg "
2010                                                                    + " of 'foreach' or 'if' or 'while' (1)");
2011                                            }
2012    
2013                                            String inType = toker.nextToken();
2014    
2015                                            if (inType.equals(type)) {
2016                                                    nestlevel--;
2017    
2018                                                    if (nestlevel < 1) {
2019                                                            break;
2020                                                    }
2021                                            }
2022    
2023                                            if ((!inType.equals("foreach")) && (!inType.equals("if"))
2024                                                            && (!inType.equals("while"))) {
2025                                                    errprintln("PL end statement requires arg of "
2026                                                                    + "'foreach' or 'if' or 'while' (2)");
2027    
2028                                                    throw new SqlToolError(
2029                                                                    "PL end statement requires arg of "
2030                                                                                    + "'foreach' or 'if' or 'while' (2)");
2031                                            }
2032                                    }
2033                            }
2034    
2035                            pw.println(s);
2036                    }
2037    
2038                    pw.flush();
2039                    pw.close();
2040    
2041                    return tmpFile;
2042            }
