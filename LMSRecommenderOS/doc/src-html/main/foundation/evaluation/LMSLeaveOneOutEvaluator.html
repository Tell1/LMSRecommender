009    package main.foundation.evaluation;
010    
011    import java.util.ArrayList;
012    import java.util.Date;
013    
014    import jcolibri.cbraplications.StandardCBRApplication;
015    import jcolibri.cbrcore.CBRCase;
016    import jcolibri.cbrcore.CBRCaseBase;
017    import jcolibri.exception.ExecutionException;
018    
019    import org.apache.commons.logging.LogFactory;
020    
021    /**
022     * This methods uses all the cases as queries. 
023     * It executes so cycles as cases in the case base. 
024     * In each cycle one case is used as query.  
025     * 
026     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
027     *         Mueller-Pettenpohl</a>
028     * 
029     * @version 2.1
030     */
031    public class LMSLeaveOneOutEvaluator extends LMSEvaluator {
032    
033            protected StandardCBRApplication app;
034    
035            @Override
036            public void init(StandardCBRApplication cbrApp) {
037    
038                    report = new LMSEvaluationReport();
039                    app = cbrApp;
040                    try {
041                            app.configure();
042                    } catch (ExecutionException e) {
043                            LogFactory.getLog(this.getClass()).error(e);
044                    }
045            }
046    
047            /**
048             * Performs the Leave-One-Out evaluation. 
049             * For each case in the case base,  remove that case from the case base and use it as query for that cycle.
050             */
051            public void LeaveOneOut() {
052                    try {
053                            java.util.ArrayList<CBRCase> aux = new java.util.ArrayList<CBRCase>();
054    
055                            long t = (new Date()).getTime();
056                            int numberOfCycles = 0;
057    
058                            // Run the precycle to load the case base
059                            LogFactory.getLog(this.getClass()).info("Running precycle()");
060                            CBRCaseBase caseBase = app.preCycle();
061    
062                            if (!(caseBase instanceof jcolibri.casebase.CachedLinealCaseBase))
063                                    LogFactory
064                                                    .getLog(this.getClass())
065                                                    .warn(
066                                                                    "Evaluation should be executed using a cached case base");
067    
068                            
069                            ArrayList<CBRCase> cases = new ArrayList<CBRCase>(caseBase.getCases());     
070                            
071                            jcolibri.util.ProgressController.init(getClass(),"LeaveOneOut Evaluation", cases.size());
072                            
073                            //For each case in the case base
074                            for(CBRCase _case : cases) {
075                                    
076                                    //Delete the case in the case base
077                                    aux.clear();
078                                    aux.add(_case);
079                                    caseBase.forgetCases(aux);
080    
081                                    //Run the cycle
082                                    LogFactory.getLog(this.getClass()).info(
083                                                    "Running cycle() " + numberOfCycles);
084                                    app.cycle(_case);
085    
086                                    //Recover case base
087                                    caseBase.learnCases(aux);
088    
089                                    numberOfCycles++;
090                                    jcolibri.util.ProgressController.step(getClass());
091                            }
092    
093                            //Run PostCycle
094                            LogFactory.getLog(this.getClass()).info("Running postcycle()");
095                            app.postCycle();
096    
097                            jcolibri.util.ProgressController.finish(getClass());
098                            
099                            t = (new Date()).getTime() - t;
100    
101                            //complete evaluation report
102                            report.setTotalTime(t);
103                            report.setNumberOfCycles(numberOfCycles);
104    
105                    } catch (ExecutionException e) {
106                            LogFactory.getLog(this.getClass()).error(e);
107                            e.printStackTrace();
108                    }
109            }
