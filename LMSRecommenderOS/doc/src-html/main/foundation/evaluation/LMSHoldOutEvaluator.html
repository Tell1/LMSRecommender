009    package main.foundation.evaluation;
010    
011    import java.util.ArrayList;
012    import java.util.Collection;
013    import java.util.Date;
014    import java.util.List;
015    
016    import jcolibri.cbraplications.StandardCBRApplication;
017    import jcolibri.cbrcore.CBRCase;
018    import jcolibri.cbrcore.CBRCaseBase;
019    import jcolibri.exception.ExecutionException;
020    
021    import org.apache.commons.logging.LogFactory;
022    
023    /**
024     * This method splits the case base in two sets: one used for testing where each
025     * case is used as query, and another that acts as normal case base. This
026     * process is performed serveral times.
027     * 
028     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
029     *         Mueller-Pettenpohl</a>
030     * 
031     * @version 2.1
032     */
033    public class LMSHoldOutEvaluator extends LMSEvaluator {
034    
035            protected StandardCBRApplication app;
036    
037            @Override
038            public void init(StandardCBRApplication cbrApp) {
039    
040                    report = new LMSEvaluationReport();
041                    app = cbrApp;
042                    try {
043                            app.configure();
044                    } catch (ExecutionException e) {
045                            LogFactory.getLog(this.getClass()).error(e);
046                    }
047            }
048    
049            /**
050             * Performs the Hold-Out evaluation.
051             * 
052             * @param testPercent
053             *            Percent of the case base used as queries. The case base is
054             *            splited randomly in each repetition.
055             * @param repetitions
056             *            Number of repetitions.
057             */
058            public void HoldOut(int testPercent, int repetitions) {
059                    try {
060                            // Obtain the time
061                            long t = (new Date()).getTime();
062                            int numberOfCycles = 0;
063                            // Run the precycle to load the case base
064                            LogFactory.getLog(this.getClass()).info("Running precycle()");
065                            CBRCaseBase caseBase = app.preCycle();
066    
067                            if (!(caseBase instanceof jcolibri.casebase.CachedLinealCaseBase))
068                                    LogFactory
069                                                    .getLog(this.getClass())
070                                                    .warn("Evaluation should be executed using a cached case base");
071    
072                            ArrayList<CBRCase> originalCases = new ArrayList<CBRCase>(
073                                            caseBase.getCases());
074    
075                            int totalSteps = ((originalCases.size() * testPercent) / 100);
076                            totalSteps = totalSteps * repetitions;
077                            jcolibri.util.ProgressController.init(getClass(),
078                                            "Hold Out Evaluation", totalSteps);
079    
080                            // For each repetition
081                            for (int rep = 0; rep < repetitions; rep++) {
082                                    ArrayList<CBRCase> querySet = new ArrayList<CBRCase>();
083                                    ArrayList<CBRCase> caseBaseSet = new ArrayList<CBRCase>();
084                                    // Split the case base
085                                    splitCaseBase(originalCases, querySet, caseBaseSet, testPercent);
086    
087                                    // Clear the caseBase
088                                    caseBase.forgetCases(originalCases);
089    
090                                    // Set the cases that acts as case base in this repetition
091                                    caseBase.learnCases(caseBaseSet);
092    
093                                    // Run cycle for each case in querySet
094                                    for (CBRCase c : querySet) {
095                                            // Run the cycle
096                                            LogFactory.getLog(this.getClass()).info(
097                                                            "Running cycle() " + numberOfCycles);
098                                            app.cycle(c);
099    
100                                            jcolibri.util.ProgressController.step(getClass());
101                                            numberOfCycles++;
102                                    }
103                            }
104    
105                            jcolibri.util.ProgressController.finish(getClass());
106    
107                            // Revert case base to original state
108                            caseBase.forgetCases(originalCases);
109                            caseBase.learnCases(originalCases);
110    
111                            // Run the poscycle to finish the application
112                            LogFactory.getLog(this.getClass()).info("Running postcycle()");
113                            app.postCycle();
114    
115                            t = (new Date()).getTime() - t;
116    
117                            // Obtain and complete the evaluation result
118                            report.setTotalTime(t);
119                            report.setNumberOfCycles(numberOfCycles);
120    
121                    } catch (Exception e) {
122                            LogFactory.getLog(this.getClass()).error(e);
123                    }
124    
125            }
126    
127            /**
128             * Splits the case base in two sets: queries and case base
129             * 
130             * @param holeCaseBase
131             *            Complete original case base
132             * @param querySet
133             *            Output param where queries are stored
134             * @param casebaseSet
135             *            Output param where case base is stored
136             * @param testPercent
137             *            Percentage of cases used as queries
138             */
139            protected void splitCaseBase(Collection<CBRCase> holeCaseBase,
140                            List<CBRCase> querySet, List<CBRCase> casebaseSet, int testPercent) {
141                    querySet.clear();
142                    casebaseSet.clear();
143    
144                    int querySetSize = (holeCaseBase.size() * testPercent) / 100;
145                    casebaseSet.addAll(holeCaseBase);
146    
147                    for (int i = 0; i < querySetSize; i++) {
148                            int random = (int) (Math.random() * casebaseSet.size());
149                            CBRCase _case = casebaseSet.get(random);
150                            casebaseSet.remove(random);
151                            querySet.add(_case);
152                    }
153            }
154    }
