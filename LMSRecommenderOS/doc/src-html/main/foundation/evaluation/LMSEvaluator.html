001    package main.foundation.evaluation;
002    
003    import jcolibri.cbraplications.StandardCBRApplication;
004    import jcolibri.evaluation.Evaluator;
005    
006    /**
007     * This abstract class defines the common behaviour of an evaluator.
008     * 
009     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
010     *         Mueller-Pettenpohl</a>
011     * 
012     * @version 1.0
013     * 
014     */
015    public abstract class LMSEvaluator extends Evaluator {
016    
017            public static final double MAX_CYCLES = 300;
018    
019            /**
020             * Initializes the evaluator with the CBR application to evaluate
021             * 
022             * @see jcolibri.cbraplications.StandardCBRApplication
023             * */
024            @Override
025            public abstract void init(StandardCBRApplication cbrApp);
026    
027            /** Object that stores the evaluation results */
028            protected static LMSEvaluationReport report;
029    
030            /** Returns the evaluation report */
031            public static LMSEvaluationReport getEvaluationReport() {
032                    return report;
033            }
034    
035            /** Sets the evaluation report */
036            public static void setEvaluationReport(LMSEvaluationReport report) {
037                    LMSEvaluator.report = report;
038            }
039    
040            public static double getMaxCycles() {
041                    return MAX_CYCLES;
042            }
043    
044            /** This evaluator can not handle more then 3000 evaluations (data sets). */
045            public static boolean isEvalLimit(int evalCount, double numCycles) {
046                    return numCycles > MAX_CYCLES ? (evalCount % Math.round(numCycles
047                                    / MAX_CYCLES)) != 0 : numCycles == MAX_CYCLES;
048            }
049    }
