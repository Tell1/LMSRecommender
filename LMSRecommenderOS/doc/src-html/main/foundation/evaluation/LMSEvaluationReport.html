001    package main.foundation.evaluation;
002    
003    import java.util.Collection;
004    import java.util.Enumeration;
005    import java.util.Hashtable;
006    import java.util.Set;
007    import java.util.Vector;
008    
009    import jcolibri.evaluation.EvaluationReport;
010    import jcolibri.method.retrieve.RetrievalResult;
011    
012    /**
013     * This class stores the result of an evaluation. It is configured and filled by
014     * an Evaluator. This info is also used to represent graphically the result of
015     * an evaluation. The stored information can be:
016     * <ul>
017     * <li>Several series of data. The lengh of the series is the number of executed
018     * cycles. And several series can be stored using diferent labels.
019     * <li>Any other information. The put/getOtherData() methods allow to store any
020     * other kind of data.
021     * <li>Number of cycles.
022     * <li>Total evaluation time.
023     * <li>Time per cycle.
024     * </ul>
025     * 
026     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
027     *         Mueller-Pettenpohl</a>
028     * 
029     * @version 2.1
030     */
031    public class LMSEvaluationReport extends EvaluationReport {
032            private long totalTime;
033            private int numberOfCycles;
034    
035            /** Stores the series info */
036            protected Hashtable<String, Vector<Double>> data;
037    
038            /** Stores other info */
039            protected Hashtable<String, String> other;
040    
041            /** Default constructor */
042            public LMSEvaluationReport() {
043                    data = new Hashtable<String, Vector<Double>>();
044                    other = new Hashtable<String, String>();
045            }
046    
047            /**
048             * Gets the evaluation info identified by the label
049             * 
050             * @param label
051             *            Identifies the evaluation serie.
052             */
053            @Override
054            public Vector<Double> getSeries(String label) {
055                    return data.get(label);
056            }
057    
058            /**
059             * Sets the evaluation info
060             * 
061             * @param label
062             *            Identifier of the info
063             * @param evaluation
064             *            Evaluation result
065             */
066            @Override
067            public void setSeries(String label, Vector<Double> evaluation) {
068    
069                    data.put(label, evaluation);
070            }
071    
072            @Override
073            public void addDataToSeries(String label, Double value) {
074                    Vector<Double> v = data.get(label);
075                    if (v == null) {
076                            v = new Vector<Double>();
077                            data.put(label, v);
078                    }
079                    v.add(value);
080            }
081    
082            public void addDataToSeries(String label, Collection<RetrievalResult> values) {
083                    Vector<Double> v = data.get(label);
084                    if (v == null) {
085                            v = new Vector<Double>();
086                            data.put(label, v);
087                    }
088                    Double avgEval = calcAverageEval(values);
089                    v.add(avgEval);
090            }
091    
092            public Double calcAverageEval(Collection<RetrievalResult> evals) {
093                    Double sumEval = 0.0;
094                    for (RetrievalResult eval : evals) {
095                            sumEval += eval.getEval();
096                    }
097                    if (sumEval.equals(Double.NaN)) {
098                            sumEval = 0.0;
099                    }
100                    return evals.size() != 0 ? sumEval / evals.size() : 0.0;
101            }
102    
103            /**
104             * Returns the name of the contained evaluation series
105             */
106            @Override
107            public String[] getSeriesLabels() {
108                    Set<String> set = data.keySet();
109                    String[] res = new String[set.size()];
110                    int i = 0;
111                    for (String e : set)
112                            res[i++] = e;
113                    return res;
114            }
115    
116            @Override
117            public void putOtherData(String label, String data) {
118                    this.other.put(label, data);
119            }
120    
121            @Override
122            public String getOtherData(String label) {
123                    return this.other.get(label);
124            }
125    
126            @Override
127            public String[] getOtherLabels() {
128                    Set<String> set = other.keySet();
129                    String[] res = new String[set.size()];
130                    int i = 0;
131                    for (String e : set)
132                            res[i++] = e;
133                    return res;
134            }
135    
136            @Override
137            public int getNumberOfCycles() {
138                    return numberOfCycles;
139            }
140    
141            @Override
142            public void setNumberOfCycles(int numberOfCycles) {
143                    this.numberOfCycles = numberOfCycles;
144            }
145    
146            @Override
147            public long getTotalTime() {
148                    return totalTime;
149            }
150    
151            @Override
152            public void setTotalTime(long totalTime) {
153                    this.totalTime = totalTime;
154            }
155    
156            @Override
157            public double getTimePerCycle() {
158                    return (double) this.totalTime / (double) numberOfCycles;
159            }
160    
161            /**
162             * Checks if the evaluation series are correct. This is: all them must have
163             * the same length
164             */
165            @Override
166            public boolean checkData() {
167                    boolean ok = true;
168                    int l = -1;
169                    for (Enumeration<Vector<Double>> iter = data.elements(); iter
170                                    .hasMoreElements() && ok;) {
171                            Vector<Double> v = iter.nextElement();
172                            if (l == -1)
173                                    l = v.size();
174                            else
175                                    ok = (l == v.size());
176                    }
177                    return ok;
178            }
179    
180            @Override
181            public String toString() {
182                    Double totalSimilarity = 0.0;
183                    StringBuffer s = new StringBuffer();
184                    s.append("Series:\n");
185                    String[] series = this.getSeriesLabels();
186                    for (int i = 0; i < series.length; i++) {
187                            s.append("  " + series[i] + ": \n    ");
188                            Vector<Double> v = this.getSeries(series[i]);
189                            for (Double d : v) {
190                                    totalSimilarity += d;
191                                    s.append("\t" + d + "\n");
192                            }
193                            totalSimilarity /= v.size();
194                            s.append("\nTotal " + series[i] + ":\n\t" + totalSimilarity);
195                    }
196    
197                    s.append("\nOther data:\n");
198                    String[] other = this.getOtherLabels();
199                    for (int i = 0; i < other.length; i++) {
200                            s.append("  " + other[i] + ": " + this.getOtherData(other[i])
201                                            + "\n");
202                    }
203    
204                    s.append("\nNumber of Cycles: \n\t" + this.getNumberOfCycles());
205                    s.append("\nTime per Cycle:   \n\t" + this.getTimePerCycle() + " ms");
206                    s.append("\nTotal time:       \n\t" + this.getTotalTime() + " ms");
207    
208                    return s.toString();
209            }
210    
211    }
