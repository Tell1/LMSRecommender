001    package main.foundation.dataTypes;
002    
003    import java.util.HashSet;
004    import java.util.Set;
005    
006    import main.domain.CMS.Document;
007    
008    /**
009     * Class that represents a set of documents and extends the TypeSet class to be
010     * able to used in the connecters of jcolibri.
011     * 
012     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
013     *         Mueller-Pettenpohl</a>
014     * 
015     * @version 1.0
016     * 
017     */
018    public class DocSet extends TypeSet<Document> {
019    
020            public DocSet() {
021                    this.set = new HashSet<Document>();
022            }
023    
024            public DocSet(Set<Document> doneDocs) {
025                    set = doneDocs;
026            }
027    
028            // [Document [name=Algoritmia], Document [name=Inteligencia Artificial]]
029            @Override
030            public void fromString(String data) throws Exception {
031    
032                    StringBuilder sb = new StringBuilder();
033                    char thisChar = ' ';
034                    Integer id = 1;
035                    int j = 0;
036                    for (int c = j; c < data.length(); c++) {
037                            thisChar = data.charAt(c);
038                            if (thisChar == '=') {
039                                    for (j = c + 1; thisChar != ']'; j++) {
040                                            thisChar = data.charAt(j);
041                                            if (thisChar != '[' && thisChar != ']') {
042                                                    sb.append(thisChar);
043                                            }
044                                    }
045                                    set.add(new Document(id++, sb.toString()));
046                                    sb = new StringBuilder();
047                            }
048                    }
049            }
050    
051    }
