001    package main.foundation.dataTypes;
002    
003    import java.util.AbstractSet;
004    import java.util.ConcurrentModificationException;
005    import java.util.HashSet;
006    import java.util.Iterator;
007    import java.util.Set;
008    
009    import jcolibri.connector.TypeAdaptor;
010    import main.domain.CMS.LearningObject;
011    
012    /**
013     * Abstract class that represents a set extending the TypeAdaptor class to be
014     * able to used in the connecters of jcolibri.
015     * 
016     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
017     *         Mueller-Pettenpohl</a>
018     * 
019     * @version 1.0
020     */
021    public abstract class TypeSet<E extends LearningObject> extends AbstractSet<E>
022                    implements TypeAdaptor {
023    
024            Set<E> set = new HashSet<E>();
025    
026            public TypeSet(Set<E> set) {
027                    super();
028                    this.set = set;
029            }
030    
031            public TypeSet() {
032                    super();
033            }
034    
035            public Set<E> getSet() {
036                    return set;
037            }
038    
039            public void setSet(Set<E> set) {
040                    this.set = set;
041            }
042    
043            @Override
044            public int hashCode() {
045                    final int prime = 31;
046                    int result = 1;
047                    result = prime * result + ((set == null) ? 0 : set.hashCode());
048                    return result;
049            }
050    
051            @Override
052            public boolean equals(Object obj) {
053                    if (this == obj)
054                            return true;
055                    if (obj == null)
056                            return false;
057                    if (getClass() != obj.getClass())
058                            return false;
059                    TypeSet<?> other = (TypeSet<?>) obj;
060                    if (set == null) {
061                            if (other.set != null)
062                                    return false;
063                    } else if (other.set.isEmpty()) {
064                            return false;
065                    } else if (!set.contains(other.set.iterator().next()))
066                            return false;
067                    return true;
068            }
069    
070            /**
071             * Returns an iterator over the elements in this set. The elements are
072             * returned in no particular order.
073             * 
074             * @return an Iterator over the elements in this set
075             * @see ConcurrentModificationException
076             */
077            @Override
078            public Iterator<E> iterator() {
079                    return set.iterator();
080            }
081    
082            /**
083             * Returns the number of elements in this set (its cardinality).
084             * 
085             * @return the number of elements in this set (its cardinality)
086             */
087            @Override
088            public int size() {
089                    return set.size();
090            }
091    
092            @Override
093            public boolean add(E e) {
094                    return set.add(e);
095            }
096    
097            @Override
098            public String toString() {
099                    return set.toString();
100            }
101    
102    }
