001    package main.foundation.dataGenerator;
002    
003    import java.io.BufferedReader;
004    import java.io.BufferedWriter;
005    import java.io.File;
006    import java.io.FileWriter;
007    import java.io.IOException;
008    import java.io.InputStreamReader;
009    import java.util.HashMap;
010    import java.util.StringTokenizer;
011    
012    import main.domain.cases.DocumentDescription;
013    import main.domain.cases.DocumentSolution;
014    import main.domain.cases.RatingResult;
015    import main.domain.cases.DocumentDescription.Courses;
016    import main.domain.cases.DocumentDescription.Subjects;
017    
018    import jcolibri.util.FileIO;
019    
020    /**
021     * Class that generates the document cases database in a proper format.
022     * 
023     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
024     *         Mueller-Pettenpohl</a>
025     * 
026     * @version 1.0
027     */
028    public class LearningObjectDocDataBaseCreater {
029            private static HashMap<Integer, DocumentDescription> docDescriptions = new HashMap<Integer, DocumentDescription>();
030            private static HashMap<Integer, DocumentSolution> docSolutions = new HashMap<Integer, DocumentSolution>();
031    
032            private static void parseDescriptions(String filename) throws Exception {
033                    BufferedReader br = null;
034                    br = new BufferedReader(
035                                    new InputStreamReader(FileIO.openFile(filename)));
036                    String line = "";
037                    while ((line = br.readLine()) != null) {
038                            StringTokenizer st = new StringTokenizer(line, "|");
039                            DocumentDescription docDesc = new DocumentDescription();
040                            docDesc.setId(Integer.parseInt(st.nextToken()));
041                            docDesc.setCourse(Courses.valueOf(st.nextToken()));
042                            docDesc.setSubject(Subjects.valueOf(st.nextToken()));
043                            docDesc.setDifficulty(Integer.parseInt(st.nextToken()));
044                            docDesc.setKeywords(st.nextToken());
045    
046                            docDescriptions.put(docDesc.getId(), docDesc);
047                    }
048                    br.close();
049    
050            }
051    
052            private static void parseSolutions(String filename) throws Exception {
053                    BufferedReader br = null;
054                    br = new BufferedReader(
055                                    new InputStreamReader(FileIO.openFile(filename)));
056    
057                    String line = "";
058                    while ((line = br.readLine()) != null) {
059                            StringTokenizer st = new StringTokenizer(line, "|");
060                            DocumentSolution docSol = new DocumentSolution();
061                            docSol.setId(Integer.parseInt(st.nextToken()));
062                            docSol.setDocName(st.nextToken());
063                            docSol.setDocAuthor(st.nextToken());
064                            docSol.setDocDate(st.nextToken());
065                            docSol.setDifficulty(Integer.parseInt(st.nextToken()));
066    
067                            docSolutions.put(docSol.getId(), docSol);
068                    }
069                    br.close();
070            }
071    
072            private static String getDescription(DocumentDescription docDesc,
073                            String sep, String quote) {
074                    return docDesc.getId() + sep + quote + docDesc.getCourse() + quote
075                                    + sep + quote + docDesc.getSubject() + quote + sep
076                                    + docDesc.getDifficulty() + sep + quote + docDesc.getKeywords()
077                                    + quote;
078            }
079    
080            private static String getSolution(DocumentSolution docSol, String sep,
081                            String quote) {
082                    return docSol.getId() + sep + quote + docSol.getDocName() + quote + sep
083                                    + quote + docSol.getDocAuthor() + quote + sep + quote
084                                    + docSol.getDocDate() + quote + sep + docSol.getDifficulty();
085            }
086    
087            private static void generateNewFile(String ratingsFile, String filename,
088                            String separator, String quote) throws Exception {
089                    BufferedWriter bw = null;
090                    bw = new BufferedWriter(new FileWriter(new File(filename), false));
091                    BufferedReader br = null;
092                    br = new BufferedReader(new InputStreamReader(
093                                    FileIO.openFile(ratingsFile)));
094                    int ratingId = 1;
095                    String line = "";
096                    while ((line = br.readLine()) != null) {
097                            StringTokenizer st = new StringTokenizer(line, "\t");
098                            Integer docId = Integer.parseInt(st.nextToken());
099                            Integer documentId = Integer.parseInt(st.nextToken());
100                            Integer rat = Integer.parseInt(st.nextToken());
101    
102                            DocumentDescription docDesc = docDescriptions.get(docId);
103                            DocumentSolution docSol = docSolutions.get(documentId);
104                            RatingResult rating = new RatingResult();
105                            rating.setId(ratingId++);
106                            rating.setRating(rat);
107    
108                            bw.write(getDescription(docDesc, separator, quote) + separator
109                                            + getSolution(docSol, separator, quote) + separator
110                                            + rating.getId() + separator + rating.getRating());
111                            bw.newLine();
112                    }
113    
114                    br.close();
115                    bw.close();
116            }
117    
118            private static void generateQuotedFile(String docfile, String filename,
119                            String separator, String quote) throws IOException {
120                    BufferedWriter bw = null;
121                    bw = new BufferedWriter(new FileWriter(new File(filename), false));
122                    BufferedReader br = null;
123                    br = new BufferedReader(new InputStreamReader(FileIO.openFile(docfile)));
124                    // int ratingId = 1;
125                    String line = "";
126                    while ((line = br.readLine()) != null) {
127                            StringTokenizer st = new StringTokenizer(line, "\t");
128                            Integer docId = Integer.parseInt(st.nextToken());
129                            Integer documentId = Integer.parseInt(st.nextToken());
130                            // Integer rat = Integer.parseInt(st.nextToken());
131    
132                            DocumentDescription docDesc = docDescriptions.get(docId);
133                            DocumentSolution docSol = docSolutions.get(documentId);
134                            // RatingResult rating = new RatingResult();
135                            // rating.setId(ratingId++);
136                            // rating.setRating(rat);
137    
138                            bw.write(getDescription(docDesc, separator, quote) + separator
139                                            + getSolution(docSol, separator, quote));
140                            bw.newLine();
141                    }
142    
143                    br.close();
144                    bw.close();
145            }
146    
147            /**
148             * @param args
149             */
150            public static void main(String[] args) {
151                    try {
152                            parseDescriptions("main/foundation/data/u.documentDescription");
153                            parseSolutions("main/foundation/data/u.documentSolution");
154    
155                            generateNewFile("main/foundation/data/u.ratingResult",
156                                            "src/main/foundation/data/documentCasesQuotedRated.txt",
157                                            "|", "'");
158                            generateQuotedFile("main/foundation/data/u.docDescDocSol",
159                                            "src/main/foundation/data/documentCasesQuoted.txt", "|",
160                                            "'");
161                    } catch (Exception e) {
162                            org.apache.commons.logging.LogFactory.getLog(
163                                            LearningObjectDocDataBaseCreater.class).error(e, e);
164    
165                    }
166    
167                    System.out
168                                    .println("File created: documentCasesQuotes.txt and documentCasesQuotedRated.txt");
169            }
