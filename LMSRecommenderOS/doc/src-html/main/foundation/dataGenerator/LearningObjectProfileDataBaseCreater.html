001    package main.foundation.dataGenerator;
002    
003    import java.io.BufferedReader;
004    import java.io.BufferedWriter;
005    import java.io.File;
006    import java.io.FileWriter;
007    import java.io.InputStreamReader;
008    import java.util.HashMap;
009    import java.util.StringTokenizer;
010    
011    import main.domain.cases.DocumentSolution;
012    import main.domain.cases.ProfileDescription;
013    import main.domain.cases.RatingResult;
014    import main.domain.cases.ProfileDescription.Degree;
015    
016    import jcolibri.util.FileIO;
017    
018    /**
019     * Class that generates the learning object database in a proper format.
020     * 
021     * @author <a href="mailto:mueller_pettenpohl@me.com">Tell
022     *         Mueller-Pettenpohl</a>
023     * 
024     * @version 1.0
025     */
026    public class LearningObjectProfileDataBaseCreater {
027            private static HashMap<Integer, ProfileDescription> profiles = new HashMap<Integer, ProfileDescription>();
028            private static HashMap<Integer, DocumentSolution> documents = new HashMap<Integer, DocumentSolution>();
029    
030            private static void parseProfiles(String filename) throws Exception {
031                    BufferedReader br = null;
032                    br = new BufferedReader(
033                                    new InputStreamReader(FileIO.openFile(filename)));
034                    String line = "";
035                    while ((line = br.readLine()) != null) {
036                            StringTokenizer st = new StringTokenizer(line, "|");
037                            ProfileDescription profile = new ProfileDescription();
038                            profile.setId(Integer.parseInt(st.nextToken()));
039                            profile.setAge(Integer.parseInt(st.nextToken()));
040                            profile.setDegree(Degree.valueOf(st.nextToken()));
041                            profile.setZipCode(st.nextToken());
042                            profile.setAvgGrade(Integer.parseInt(st.nextToken()));
043                            profile.setAvgDifficulty(Integer.parseInt(st.nextToken()));
044                            profile.setAvgEvaluation(Integer.parseInt(st.nextToken()));
045    
046                            profiles.put(profile.getId(), profile);
047                    }
048                    br.close();
049    
050            }
051    
052            private static void parseDocuments(String filename) throws Exception {
053                    BufferedReader br = null;
054                    br = new BufferedReader(
055                                    new InputStreamReader(FileIO.openFile(filename)));
056    
057                    String line = "";
058                    while ((line = br.readLine()) != null) {
059                            StringTokenizer st = new StringTokenizer(line, "|");
060                            DocumentSolution document = new DocumentSolution();
061                            document.setId(Integer.parseInt(st.nextToken()));
062                            document.setDocName(st.nextToken());
063                            document.setDocAuthor(st.nextToken());
064                            document.setDocDate(st.nextToken());
065                            document.setDifficulty(Integer.parseInt(st.nextToken()));
066    
067                            documents.put(document.getId(), document);
068                    }
069                    br.close();
070            }
071    
072            private static String getProfile(ProfileDescription profile, String sep) {
073                    return profile.getId() + sep + profile.getAge() + sep
074                                    + profile.getDegree() + sep + profile.getZipCode() + sep
075                                    + profile.getAvgGrade() + sep + profile.getAvgDifficulty()
076                                    + sep + profile.getAvgEvaluation();
077            }
078    
079            private static String getDocument(DocumentSolution document, String sep) {
080                    return document.getId() + sep + document.getDocName() + sep
081                                    + document.getDocAuthor() + sep + document.getDocDate() + sep
082                                    + document.getDifficulty();
083            }
084    
085            private static void generateNewFile(String ratingsFile, String filename,
086                            String separator) throws Exception {
087                    BufferedWriter bw = null;
088                    bw = new BufferedWriter(new FileWriter(new File(filename), false));
089                    BufferedReader br = null;
090                    br = new BufferedReader(new InputStreamReader(
091                                    FileIO.openFile(ratingsFile)));
092                    int ratingId = 1;
093                    String line = "";
094                    while ((line = br.readLine()) != null) {
095                            StringTokenizer st = new StringTokenizer(line, "\t");
096                            Integer profileId = Integer.parseInt(st.nextToken());
097                            Integer documentId = Integer.parseInt(st.nextToken());
098                            Integer rat = Integer.parseInt(st.nextToken());
099    
100                            ProfileDescription profile = profiles.get(profileId);
101                            DocumentSolution document = documents.get(documentId);
102                            RatingResult rating = new RatingResult();
103                            rating.setId(ratingId++);
104                            rating.setRating(rat);
105    
106                            bw.write(getProfile(profile, separator) + separator
107                                            + getDocument(document, separator) + separator
108                                            + rating.getId() + separator + rating.getRating());
109                            bw.newLine();
110                    }
111    
112                    br.close();
113                    bw.close();
114            }
115    
116            /**
117             * @param args
118             */
119            public static void main(String[] args) {
120                    try {
121                            parseProfiles("main/foundation/data/u.profileDescription");
122                            parseDocuments("main/foundation/data/u.documentSolution");
123    
124                            generateNewFile("main/foundation/data/u.ratingResult",
125                                            "src/main/foundation/data/learningObjectCases.txt", "|");
126                    } catch (Exception e) {
127                            org.apache.commons.logging.LogFactory.getLog(
128                                            LearningObjectProfileDataBaseCreater.class).error(e, e);
129    
130                    }
131    
132                    System.out.println("File created: learningObjectCases.txt");
133    
134            }
135    
136    }
